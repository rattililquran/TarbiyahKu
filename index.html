-- ── LANGKAH 1 — Sesuaikan fungsi yang menyentuh jurnal (create or replace) ── -- 1a. initial_data tanpa key 'journals' (harus lebih dulu sebelum drop app_journals) create or replace function public.app_get_initial_data() returns jsonb language plpgsql stable security definer set search_path = public as $$ declare uid uuid := public._uid(); begin return jsonb_build_object( 'profile', public._profil_json(uid), 'todayProgress',public.app_today_progress() ); end $$; -- 1b. reset_reading tanpa delete jurnal (harus lebih dulu sebelum drop table) create or replace function public.app_reset_reading() returns jsonb language plpgsql security definer set search_path = public as $$ declare uid uuid := public._uid(); begin update public.progres set tanggal_target=null, target_harian=0, halaman_terakhir=0, surah_terakhir=1, ayat_terakhir=0, streak=0, tanggal_update=null, total_khatam=0, status_udzur='', udzur_start=null, udzur_end=null where user_id = uid; if not found then raise exception 'Profil pengguna tidak ditemukan.'; end if; delete from public.log_bacaan where user_id = uid; return public._profil_json(uid); end $$; -- ── LANGKAH 2 — Drop RPC jurnal ── drop function if exists public.app_save_journal(jsonb);
